TrmnlApi response — https://trmnl-plugins-api.azurewebsites.net/api/v1/forecast

Field names are unit-agnostic. Actual units depend on the `units` query parameter:
    imperial (default): °F, mph, inches
    metric:             °C, km/h, mm

Query parameters:
    latitude    (required) — location latitude
    longitude   (required) — location longitude
    units       — "imperial" (default) or "metric"
    hours       — number of hourly entries (1–25, default 25)
    days        — number of daily entries (1–6, default 6)
    provider    — "pirate-weather" (default) or "open-meteo"
    fake        — "true" or "1" to inject randomized precipitation (testing)

data.
    current.
        time                    (ISO 8601, e.g. "2026-02-28T14:00")
        temperature             (integer, °F or °C)
        apparent_temperature    (integer, "feels like", °F or °C)
        relative_humidity       (%, 0-100)
        precipitation           (inch or mm)
        condition               (string, e.g. "Clear", "Overcast", "Rain")
        icon_class              (CSS class, e.g. "wi-day-sunny", "wi-night-clear")
        wind_speed              (integer, mph or km/h)
        wind_direction_deg      (degrees, 0–360)
        wind_direction          (cardinal string, e.g. "N", "SW")
        is_day                  (boolean)
    hourly.
        entries                 (array of up to 25 objects, current hour + next 24h)
            time                (ISO 8601)
            label               (string, e.g. "Now", "3pm")
            temperature         (integer)
            precipitation_probability (%, 0–100)
            icon_class          (CSS class, day/night variant pre-computed)
            is_day              (boolean)
    daily.
        entries                 (array of up to 6 objects, today + next 5 days)
            date                (date string, e.g. "2026-02-28")
            high                (integer, daily high temperature)
            low                 (integer, daily low temperature)
            condition           (string, e.g. "Clear", "Snow")
            icon_class          (CSS class, no day/night variant for daily)
            precipitation_probability (%, 0–100)
            sunrise             (ISO 8601, e.g. "2026-02-28T06:20")
            sunset              (ISO 8601, e.g. "2026-02-28T17:32")
    meta.
        cache                   (string: "fresh_fetch", "fresh_hit", or "stale_served")
        provider                (string: provider that actually served this response, e.g. "pirate-weather" or "open-meteo"; may differ from requested_provider on fallback)
        requested_provider      (string: provider the caller asked for)
        fetched_at              (ISO 8601 UTC timestamp of last upstream fetch)
        data_time               (ISO 8601, matches current.time)
        served_at               (ISO 8601 UTC timestamp when this response was served)
        age_seconds             (integer, seconds since fetched_at)
        upstream                (null when healthy; { status, error } when stale_served or when a fallback was used)
